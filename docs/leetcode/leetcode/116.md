## 解法一

层次遍历

```js
var connect = function(root) {
  if (!root) return null
  const queue = []
  queue.push(root)
  let prevsbiling = queue[queue.length - 1]
  while (queue.length) {
    let n = queue.length
    while (n--) {
      let node = queue.pop()
      if (node.left) queue.unshift(node.left)
      if (node.right) queue.unshift(node.right)

      if (n) {
        prevsbiling.next = node
        prevsbiling = node
      } else {
        prevsbiling.next = node
        node.next = null
        prevsbiling = queue[queue.length - 1]
      }
    }
  }
  return root
}
```

## 解法二

层次遍历优化版

```js
var connect = function(root) {
  let leftNode = root
  while (leftNode.left) {
    head = leftNode
    leftNode = head.left
    while (head) {
      head.left.next = head.right
      if (head.next) {
        head.right = head.next.left
      }
      head = head.next
    }
  }
  return root
}
```
